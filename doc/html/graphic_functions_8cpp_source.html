34void drawTiles(sf::RenderWindow &window, const int tileSize, const int windowWidth, const int windowHeight,int difficulty) {
35 //Count how many tiles we can fit in map
36 const int mapWidth = windowWidth / tileSize;
37 const int mapHeight = windowHeight / tileSize;
38
39
40 // Define a hardcoded map, 0 for water(blue), 1 for grass(green) and other for path(white)
41 int map[16][12];
42 std::ifstream mapFile;
43 if(difficulty > 3){
44 mapFile.open("src/assets/map2.txt");
45 }
46 else{
47 mapFile.open("src/assets/map1.txt");
48 }
49
50 if (!mapFile) {
51 std::cerr << "Unable to open map file";
52 // handle error
53 }
54
55 for (int i = 0; i < 16; ++i) {
56 for (int j = 0; j < 12; ++j) {
57 char ch;
58 if (!(mapFile >> ch)) {
59 //std::cerr << "Error reading map file";
60 // handle error
61 }
62 map[i][j] = ch - '0'; // convert char to int
63 }
64 }
65
66 mapFile.close();
67
68 //iterate through all the tiles
69 for (int x = 0; x < mapWidth; x++) {
70 for (int y = 0; y < mapHeight; y++) {
71 //calculate position
72 sf::Vector2f tilePosition(x * tileSize, y * tileSize);
73
74 // Create a Tile object with the specified color, and tileSize
75 int tileType = map[x][y];
76 sf::Color tileColor;
77 if (tileType == 0) {
78 tileColor = sf::Color::Blue;
79 } else if (tileType == 1) {
80 tileColor = sf::Color::Green;
81 } else {
82 tileColor = sf::Color::White;
83 }
84 //create tile objects for each tile
85 Tile tile(tilePosition, tileColor, tileSize);
86
87 tiles.push_back(tile);
88 // Draw the tile's shape
89 window.draw(tile.getShape());
90 }
91 }
